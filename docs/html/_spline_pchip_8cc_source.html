52  REGION_ABCDEM
53  get_region( real_type alpha, real_type beta ) {
54  // assuming alpha >= 0, beta >= 0
55  real_type apb = alpha+beta;
56  if ( apb <= 3 ) return region_M;
57  real_type r = alpha * alpha + beta * beta + alpha * beta - 6*apb + 9;
58  if ( r <= 0 ) return region_M;
59  if ( apb <= 4 ) {
60  if ( beta >= alpha ) return region_A;
61  else return region_E;
62  }
63  if ( alpha >= 3 && beta >= 3 ) return region_C;
64  if ( beta >= alpha ) return region_B;
65  else return region_D;
66  }

103  int sa = 0;
104  int sb = 0;
105  if ( a > 0 ) sa = 1;
106  else if ( a < 0 ) sa = -1;

124  void
125  Pchip_build(
126  real_type const X[],
127  real_type const Y[],
128  real_type Yp[],
129  integer npts
130  ) {
131 
132  size_t n = npts > 0 ? size_t( npts - 1 ) : 0;
133 
134  integer ierr = 0;
135 
136  // function definition is ok, go on.
137  real_type h1 = X[1] - X[0];
138  real_type del1 = (Y[1]-Y[0])/h1;
139  real_type dsave = del1;
140 
141  // special case n=2 -- use linear interpolation.
142  if ( n == 1 ) { Yp[0] = Yp[1] = del1; return; }
143 
144  real_type h2 = X[2] - X[1];
145  real_type del2 = (Y[2]-Y[1])/h2;
146 
147  // Set Yp[0] via non-centered three-point formula, adjusted to be shape-preserving.
148  real_type hsum = h1 + h2;
149  real_type w1 = (h1 + hsum)/hsum;
150  real_type w2 = -h1/hsum;
151  Yp[0] = w1*del1 + w2*del2;
152  real_type dmin, dmax;
153  if ( signTest(Yp[0],del1) <= 0 ) {
154  Yp[0] = 0;
155  } else if ( signTest(del1,del2) < 0 ) {
156  // NEED DO THIS CHECK ONLY IF MONOTONICITY SWITCHES.
157  dmax = 3*del1;
158  if ( std::abs(Yp[0]) > std::abs(dmax) ) Yp[0] = dmax;
159  }
160 
161  // loop through interior points.
162  for ( size_t i = 1; i < n; ++i ) {
163  if ( i > 1 ) {
164  h1 = h2;
165  h2 = X[i+1] - X[i];
166  hsum = h1 + h2;
167  del1 = del2;
168  del2 = (Y[i+1] - Y[i])/h2;
169  }
170  // set Yp[i]=0 unless data are strictly monotonic.
171  Yp[i] = 0;
172  // count number of changes in direction of monotonicity.
173  switch ( signTest(del1,del2) ) {
174  case -1:
175  if ( Utils::isZero(del2) ) break;
176  if ( signTest(dsave,del2) < 0 ) ++ierr;
177  dsave = del2;
178  break;
179  case 0:
180  ++ierr;
181  dsave = del2;
182  break;
183  case 1: // use brodlie modification of butland formula.
184  w1 = (1+h1/hsum)/3;
185  w2 = (1+h2/hsum)/3;
186  dmax = max_abs( del1, del2 );
187  dmin = min_abs( del1, del2 );
188  real_type drat1 = del1/dmax;
189  real_type drat2 = del2/dmax;
190  Yp[i] = dmin/(w1*drat1 + w2*drat2);
191  break;
192  }
193  }
194  // set Yp[n] via non-centered three-point formula, adjusted to be shape-preserving.
195  w1 = -h2/hsum;
196  w2 = (h2 + hsum)/hsum;
197  Yp[n] = w1*del1 + w2*del2;
198  if ( signTest(Yp[n],del2) <= 0 ) {
199  Yp[n] = 0;
200  } else if ( signTest(del1,del2) < 0 ) {
201  // need do this check only if monotonicity switches.
202  dmax = 3*del2;
203  if ( abs(Yp[n]) > abs(dmax) ) Yp[n] = dmax;
204  }
205  // cout << "ierr = " << ierr << '\n';
206  }

208  static // non usata per ora
209  void
210  Pchip_build_new(
211  real_type const X[],
212  real_type const Y[],
213  real_type Yp[],
214  integer npts
215  ) {
216 
217  first_derivative_build( X, Y, Yp, npts );
218 
219  size_t n = npts > 0 ? size_t( npts - 1 ) : 0;
220 
221  // loop through interior points.
222  for ( size_t i = 1; i < n ; ++i ) {
223 
224  real_type hL = X[i+0] - X[i-1];
225  real_type hR = X[i+1] - X[i+0];
226 
227  real_type SL = (Y[i+0]-Y[i-1])/hL;
228  real_type SR = (Y[i+1]-Y[i+0])/hR;
229 
230  real_type fp = Yp[i];
231 
232  real_type sigma = 0;
233  if ( SL*SR > 0 ) sigma = SR > 0 ? 1 : -1;
234  real_type absSL = SL < 0 ? -SL: SL;
235  real_type absSR = SR < 0 ? -SR: SR;
236  real_type Delta = 3*( absSL < absSR ? absSL : absSR );
237  if ( sigma > 0 ) {
238  if ( fp < 0 ) fp = 0;
239  if ( fp > Delta ) fp = Delta;
240  } else if ( sigma < 0 ) {
241  if ( fp > 0 ) fp = 0;
242  if ( fp < -Delta ) fp = -Delta;
243  } else {
244  fp = 0;
245  }
246  Yp[i] = fp;
247  }
248  }
