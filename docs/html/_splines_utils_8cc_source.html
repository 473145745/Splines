282  real_type
283  second_deriv3p_C(
284  real_type SL,
285  real_type hL,
286  real_type SR,
287  real_type hR,
288  real_type dpL,
289  real_type dp0,
290  real_type dpR
291  ) {
292  real_type hLR = hL*hR;
293  real_type hL2 = hL*hL;
294  real_type hL4 = hL2*hL2;
295  real_type hR2 = hR*hR;
296  real_type hR4 = hR2*hR2;
297  real_type top = hL4*(6*SR-4*dp0-2*dpR) + hR4*(4*dp0+2*dpL-6*SL) +
298  hLR*( hL2*(10*SR-8*dp0-2*dpR) - hR2*(10*SL-8*dp0-2*dpL) );
299  real_type tmp = hR+hL;
300  real_type bot = tmp*tmp*tmp*hLR;
301  return top/bot;
302  }

304  real_type
305  second_deriv3p_C(
306  real_type SL,
307  real_type hL,
308  real_type SR,
309  real_type hR,
310  real_type dp0
311  ) {
312  real_type hh = hL+hR;
313  real_type alpha = hL/hh;
314  real_type beta = hR/hh;
315  real_type DDR = (SR-dp0)/hR;
316  real_type DDL = (dp0-SL)/hL;
317  return 2*(alpha*DDR+beta*DDL);
318  }

322  real_type
323  second_deriv3p_L(
324  real_type SL,
325  real_type hL,
326  real_type SR,
327  real_type hR,
328  real_type dpL,
329  real_type dp0,
330  real_type dpR
331  ) {
332  real_type hLR = hL * hR;
333  real_type hL2 = hL * hL;
334  real_type hL4 = hL2 * hL2;
335  real_type hR2 = hR * hR;
336  real_type hR4 = hR2 * hR2;
337  real_type top = hL4 * (4 * SR - 2 * (dp0+dpR) )
338  + hR4 * (6 * SL - 2 * dp0 - 4 * dpL)
339  + hL2 * hR2 * (20*SL - 12*dp0 - 8*dpL)
340  + hLR * hR2 * (20*SL - 8*dp0 - 12*dpL)
341  + hL2 * hLR * (10*SR - 8*dp0 - 2*dpR);
342  real_type hRpL = hR + hL;
343  real_type bot = hR2 * hL * (hRpL*hRpL);
344  return top/bot;
345  }

347  real_type
348  second_deriv3p_L(
349  real_type SL,
350  real_type hL,
351  real_type SR,
352  real_type hR,
353  real_type dpL
354  ) {
355  real_type hh = hL+hR;
356  real_type alpha = hL/hh;
357  real_type beta = hR/hh;
358  real_type t1 = (6*SL-2*SR-4*dpL)/hL;
359  real_type t3 = 2*(SL-dpL)/hL;
360  real_type t2 = 3*t3;
361  return t1*alpha*alpha+t2*alpha*beta+t3*beta*beta;
362  }

411  void
412  first_derivative_build(
413  real_type const X[],
414  real_type const Y[],
415  real_type Yp[],
416  integer npts
417  ) {
418 
419  size_t n = npts > 0 ? size_t( npts - 1 ) : 0;
420 
421  // special case n=2 -- use linear interpolation.
422  {
423  real_type hL = X[1] - X[0];
424  real_type SL = (Y[1] - Y[0])/hL;
425  switch ( npts ) {
426  case 2:
427  Yp[0] = Yp[1] = SL;
428  return;
429  case 3:
430  {
431  real_type hR = X[2] - X[1];
432  real_type SR = (Y[2] - Y[1])/hR;
433  Yp[0] = first_deriv3p_L( SL, hL, SR, hR );
434  Yp[1] = first_deriv3p_C( SL, hL, SR, hR );
435  Yp[2] = first_deriv3p_R( SL, hL, SR, hR );
436  }
437  return;
438  case 4:
439  {
440  real_type hC = X[2] - X[1];
441  real_type SC = (Y[2] - Y[1])/hC;
442  real_type hR = X[3] - X[2];
443  real_type SR = (Y[3] - Y[2])/hR;
444  Yp[0] = first_deriv4p_L( SL, hL, SC, hC, SR, hR );
445  Yp[1] = first_deriv3p_C( SL, hL, SC, hC );
446  Yp[2] = first_deriv3p_C( SC, hC, SR, hR );
447  Yp[3] = first_deriv4p_R( SL, hL, SC, hC, SR, hR );
448  }
449  return;
450  }
451  }
452 
453  // loop through interior points.
454  for ( size_t i = 2; i < n-1 ; ++i ) {
455 
456  real_type hLL = X[i-1] - X[i-2];
457  real_type hL = X[i+0] - X[i-1];
458  real_type hR = X[i+1] - X[i+0];
459  real_type hRR = X[i+2] - X[i+1];
460 
461  real_type SLL = (Y[i-1]-Y[i-2])/hLL;
462  real_type SL = (Y[i+0]-Y[i-1])/hL;
463  real_type SR = (Y[i+1]-Y[i+0])/hR;
464  real_type SRR = (Y[i+2]-Y[i+1])/hRR;
465 
466  Yp[i] = first_deriv5p_C( SLL, hLL, SL, hL, SR, hR, SRR, hRR );
467  }
468  {
469  real_type hL = X[n] - X[n-1];
470  real_type hLL = X[n-1] - X[n-2];
471  real_type SL = (Y[n] - Y[n-1])/hL;
472  real_type SLL = (Y[n-1] - Y[n-2])/hLL;
473  deriv_right( SL, hL, SLL, hLL, Yp[n-2], Yp[n], Yp[n-1] );
474  }
475  {
476  real_type hR = X[1] - X[0];
477  real_type hRR = X[2] - X[1];
478  real_type SR = (Y[1] - Y[0])/hR;
479  real_type SRR = (Y[2] - Y[1])/hRR;
480  deriv_left( SR, hR, SRR, hRR, Yp[2], Yp[0], Yp[1] );
481  }
482 
483  }

487  void
488  second_derivative_build(
489  real_type const X[],
490  real_type const Y[],
491  real_type const Yp[],
492  real_type Ypp[],
493  integer npts
494  ) {
495 
496  size_t n = npts > 0 ? size_t( npts - 1 ) : 0;
497 
498  // special case n=2 -- use linear interpolation.
499  switch ( npts ) {
500  case 2:
501  Ypp[0] = Ypp[1] = 0;
502  return;
503  }
504 
505  // loop through interior points.
506  for ( size_t i = 1; i < n ; ++i ) {
507 
508  real_type hL = X[i+0] - X[i-1];
509  real_type hR = X[i+1] - X[i+0];
510  real_type SL = (Y[i+0]-Y[i-1])/hL;
511  real_type SR = (Y[i+1]-Y[i+0])/hR;
512 
513  real_type dp0 = Yp[i];
514  real_type dpL = Yp[i-1];
515  real_type dpR = Yp[i+1];
516 
517  Ypp[i] = second_deriv3p_C( SL, hL, SR, hR, dp0, dpL, dpR );
518  }
519  {
520  real_type hL = X[1] - X[0];
521  real_type hR = X[2] - X[1];
522  real_type SL = (Y[1]-Y[0])/hL;
523  real_type SR = (Y[2]-Y[1])/hR;
524  real_type dp0 = Yp[1];
