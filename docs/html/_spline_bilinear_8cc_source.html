41  real_type
42  BilinearSpline::operator () ( real_type x, real_type y ) const {
43  integer i = this->search_x( x );
44  integer j = this->search_y( y );
45  real_type DX = m_X[size_t(i+1)] - m_X[size_t(i)];
46  real_type DY = m_Y[size_t(j+1)] - m_Y[size_t(j)];
47  real_type u = (x-m_X[size_t(i)])/DX;
48  real_type v = (y-m_Y[size_t(j)])/DY;
49  real_type u1 = 1-u;
50  real_type v1 = 1-v;
51  real_type Z00 = m_Z[size_t(this->ipos_C(i,j))];
52  real_type Z01 = m_Z[size_t(this->ipos_C(i,j+1))];
53  real_type Z10 = m_Z[size_t(this->ipos_C(i+1,j))];
54  real_type Z11 = m_Z[size_t(this->ipos_C(i+1,j+1))];
55  return u1 * ( Z00 * v1 + Z01 * v ) +
56  u * ( Z10 * v1 + Z11 * v );
57  }

61  real_type
62  BilinearSpline::Dx( real_type x, real_type y ) const {
63  integer i = this->search_x( x );
64  integer j = this->search_y( y );
65  real_type DX = m_X[size_t(i+1)] - m_X[size_t(i)];
66  real_type DY = m_Y[size_t(j+1)] - m_Y[size_t(j)];
67  real_type v = (y-m_Y[size_t(j)])/DY;
68  real_type Z00 = m_Z[size_t(ipos_C(i,j))];
69  real_type Z01 = m_Z[size_t(ipos_C(i,j+1))];
70  real_type Z10 = m_Z[size_t(ipos_C(i+1,j))];
71  real_type Z11 = m_Z[size_t(ipos_C(i+1,j+1))];
72  return ( (Z10-Z00) * (1-v) + (Z11-Z01) * v ) / DX;
73  }

77  real_type
78  BilinearSpline::Dy( real_type x, real_type y ) const {
79  integer i = this->search_x( x );
80  integer j = this->search_y( y );
81  real_type DX = m_X[size_t(i+1)] - m_X[size_t(i)];
82  real_type DY = m_Y[size_t(j+1)] - m_Y[size_t(j)];
83  real_type u = (x-m_X[size_t(i)])/DX;
84  real_type Z00 = m_Z[size_t(ipos_C(i,j))];
85  real_type Z01 = m_Z[size_t(ipos_C(i,j+1))];
86  real_type Z10 = m_Z[size_t(ipos_C(i+1,j))];
87  real_type Z11 = m_Z[size_t(ipos_C(i+1,j+1))];
88  return ( ( Z01-Z00 ) * (1-u) + ( Z11-Z10 ) * u ) / DY;
89  }

93  void
94  BilinearSpline::D( real_type x, real_type y, real_type d[3] ) const {
95  integer i = this->search_x( x );
96  integer j = this->search_y( y );
97  real_type DX = m_X[size_t(i+1)] - m_X[size_t(i)];
98  real_type DY = m_Y[size_t(j+1)] - m_Y[size_t(j)];
99  real_type u = (x-m_X[size_t(i)])/DX;
100  real_type v = (y-m_Y[size_t(j)])/DY;
101  real_type u1 = 1-u;
102  real_type v1 = 1-v;
103  real_type Z00 = m_Z[size_t(this->ipos_C(i,j))];
104  real_type Z01 = m_Z[size_t(this->ipos_C(i,j+1))];
105  real_type Z10 = m_Z[size_t(this->ipos_C(i+1,j))];
106  real_type Z11 = m_Z[size_t(this->ipos_C(i+1,j+1))];
107  d[0] = u1 * ( Z00 * v1 + Z01 * v ) + u * ( Z10 * v1 + Z11 * v );
108  d[1] = v1 * (Z10-Z00) + v * (Z11-Z01); d[1] /= DX;
109  d[2] = u1 * (Z01-Z00) + u * (Z11-Z10); d[2] /= DY;
110  }
