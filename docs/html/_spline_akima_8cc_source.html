49  static
50  real_type
51  akima_one(
52  real_type epsi,
53  real_type di_m2,
54  real_type di_m1,
55  real_type di,
56  real_type di_p1
57  ) {
58  real_type wl = std::abs(di_p1 - di);
59  real_type wr = std::abs(di_m1 - di_m2);
60  real_type den = wl + wr;
61  if ( den <= epsi ) { wl = wr = 0.5; den = 1; } // if epsi == 0
62  real_type num = wl * di_m1 + wr * di;
63  return num / den;
64  }

68  void
69  Akima_build(
70  real_type const X[],
71  real_type const Y[],
72  real_type Yp[],
73  integer npts
74  ) {
75 
76  if ( npts == 2 ) { // solo 2 punti, niente da fare
77  Yp[0] = Yp[1] = (Y[1]-Y[0])/(X[1]-X[0]);
78  } else {
79  Utils::Malloc<real_type> mem("Akima_build");
80  mem.allocate( size_t(npts+3) );
81  real_type * m = mem( size_t(npts+3) );
82 
83  // calcolo slopes (npts-1) intervals + 4
84  for ( size_t i = 1; i < size_t(npts); ++i )
85  m[i+1] = (Y[i]-Y[i-1])/(X[i]-X[i-1]);
86 
87  // extra slope at the boundary
88  m[1] = 2*m[2]-m[3];
89  m[0] = 2*m[1]-m[2];
90  m[size_t(npts+1)] = 2*m[size_t(npts)]-m[size_t(npts-1)];
91  m[size_t(npts+2)] = 2*m[size_t(npts+1)]-m[size_t(npts)];
92 
93  // minimum delta slope
94  real_type epsi = 0;
95  for ( size_t i = 0; i < size_t(npts+2); ++i ) {
96  real_type dm = std::abs(m[i+1]-m[i]);
97  if ( dm > epsi ) epsi = dm;
98  }
99  epsi *= 1E-8;
100 
101  // 0 1 2 3 4---- n-1 n n+1 n+2
102  // + + + + +
103  for ( size_t i = 0; i < size_t(npts); ++i )
104  Yp[i] = akima_one( epsi, m[i], m[i+1], m[i+2], m[i+3] );
105  }
106  }

110  void
111  AkimaSpline::build() {
112  UTILS_ASSERT(
113  m_npts > 1,
114  "AkimaSpline::build(): npts = {} not enought points\n", m_npts
115  );
116  integer ibegin = 0;
117  integer iend = 0;
118  do {
119  // cerca intervallo monotono strettamente crescente
120  while ( ++iend < m_npts && m_X[iend-1] < m_X[iend] ) {}
121  Akima_build( m_X+ibegin, m_Y+ibegin, m_Yp+ibegin, iend-ibegin );
122  ibegin = iend;
123  } while ( iend < m_npts );
124 
125  SPLINE_CHECK_NAN( m_Yp, "AkimaSpline::build(): Yp", m_npts );
126  }
