124  real_type
125  CubicSplineBase::id_eval( integer i, real_type x ) const {
126  if ( m_curve_can_extend && m_curve_extended_constant ) {
127  if ( x <= m_X[0] ) return m_Y[0];
128  if ( x >= m_X[m_npts-1] ) return m_Y[m_npts-1];
129  }
130  real_type base[4];
131  Hermite3( x-m_X[i], m_X[i+1]-m_X[i], base );
132  return base[0] * m_Y[i] +
133  base[1] * m_Y[i+1] +
134  base[2] * m_Yp[i] +
135  base[3] * m_Yp[i+1];
136  }

217  integer // order
218  CubicSplineBase::coeffs(
219  real_type cfs[],
220  real_type nodes[],
221  bool transpose
222  ) const {
223  size_t n = size_t(m_npts > 0 ? m_npts-1 : 0);
224  for ( size_t i = 0; i < n; ++i ) {
225  nodes[i] = m_X[i];
226  real_type H = m_X[i+1]-m_X[i];
227  real_type DY = (m_Y[i+1]-m_Y[i])/H;
228  real_type a = m_Y[i];
229  real_type b = m_Yp[i];
230  real_type c = (3*DY-2*m_Yp[i]-m_Yp[i+1])/H;
231  real_type d = (m_Yp[i+1]+m_Yp[i]-2*DY)/(H*H);
232  if ( transpose ) {
233  cfs[4*i+3] = a;
234  cfs[4*i+2] = b;
235  cfs[4*i+1] = c;
236  cfs[4*i+0] = d;
237  } else {
238  cfs[i+3*n] = a;
239  cfs[i+2*n] = b;
240  cfs[i+1*n] = c;
241  cfs[i+0*n] = d;
242  }
243  }
244  return 4;
245  }

270  Spline::setRange( xmin, xmax );
