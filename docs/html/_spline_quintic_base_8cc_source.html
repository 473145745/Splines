89  real_type
90  QuinticSplineBase::id_eval( integer i, real_type x ) const {
91  if ( m_curve_can_extend && m_curve_extended_constant ) {
92  if ( x <= m_X[0] ) return m_Y[0];
93  if ( x >= m_X[m_npts-1] ) return m_Y[m_npts-1];
94  }
95  real_type base[6];
96  real_type x0 = m_X[i];
97  real_type H = m_X[i+1] - x0;
98  Hermite5( x-x0, H, base );
99  return base[0] * m_Y[i] + base[1] * m_Y[i+1] +
100  base[2] * m_Yp[i] + base[3] * m_Yp[i+1] +
101  base[4] * m_Ypp[i] + base[5] * m_Ypp[i+1];
102  }

234  integer // order
235  QuinticSplineBase::coeffs(
236  real_type cfs[],
237  real_type nodes[],
238  bool transpose
239  ) const {
240  size_t n = size_t(m_npts > 0 ? m_npts-1 : 0);
241  for ( size_t i = 0; i < n; ++i ) {
242  nodes[i] = m_X[i];
243  real_type H = m_X[i+1]-m_X[i];
244  real_type a = m_Y[i];
245  real_type b = m_Yp[i];
246  real_type c = m_Ypp[i]/2;
247  real_type d = ((10*(m_Y[i+1]-m_Y[i])/H-6*m_Yp[i]-4*m_Yp[i+1])/H-1.5*m_Ypp[i]+0.5*m_Ypp[i+1])/H;
248  real_type e = ((15*(m_Y[i]-m_Y[i+1])/H+8*m_Yp[i]+7*m_Yp[i+1])/H+1.5*m_Ypp[i]-m_Ypp[i+1])/(H*H);
249  real_type f = ((6*(m_Y[i+1]-m_Y[i])/H-3*(m_Yp[i]+m_Yp[i+1]))/H-0.5*m_Ypp[i]+0.5*m_Ypp[i+1])/(H*H*H);
250  if ( transpose ) {
251  cfs[6*i+5] = a;
252  cfs[6*i+4] = b;
253  cfs[6*i+3] = c;
254  cfs[6*i+2] = d;
255  cfs[6*i+1] = e;
256  cfs[6*i+0] = f;
257  } else {
258  cfs[i+5*n] = a;
259  cfs[i+4*n] = b;
260  cfs[i+3*n] = c;
261  cfs[i+2*n] = d;
262  cfs[i+1*n] = e;
263  cfs[i+0*n] = f;
264  }
265  }
266  return 6;
267  }
