261  void
262  CubicSpline_build(
263  real_type const X[],
264  real_type const Y[],
265  real_type Yp[],
266  real_type Ypp[],
267  real_type L[],
268  real_type D[],
269  real_type U[],
270  integer npts,
271  CUBIC_SPLINE_TYPE_BC bc0,
272  CUBIC_SPLINE_TYPE_BC bcn
273  ) {
274 
275  size_t n = size_t(npts > 0 ? npts-1 : 0);
276  real_type * Z = Ypp;
277 
278  size_t i;
279  for ( i = 1; i < n; ++i ) {
280  real_type HL = X[i] - X[i-1];
281  real_type HR = X[i+1] - X[i];
282  real_type HH = HL+HR;
283  L[i] = HL/HH;
284  U[i] = HR/HH;
285  D[i] = 2;
286  Z[i] = 6 * ( (Y[i+1]-Y[i])/HR - (Y[i]-Y[i-1])/HL ) / HH;
287  }
288 
289  real_type UU = 0, LL = 0;
290 
291  switch ( bc0 ) {
292  case EXTRAPOLATE_BC:
293  L[0] = 0; D[0] = 1; U[0] = 0;
294  if ( npts == 2 ) {
295  Z[0] = 0;
296  } else if ( npts == 3 ) {
297  real_type hR = X[1] - X[0];
298  real_type hRR = X[2] - X[1];
299  real_type SR = (Y[1] - Y[0])/hR;
300  real_type SRR = (Y[2] - Y[1])/hRR;
301  Z[0] = deriv2_3p_L( SR, hR, SRR, hRR );
302  } else if ( npts == 4 ) {
303  real_type hR = X[1] - X[0];
304  real_type hRR = X[2] - X[1];
305  real_type hRRR = X[3] - X[2];
306  real_type SR = (Y[1] - Y[0])/hR;
307  real_type SRR = (Y[2] - Y[1])/hRR;
308  real_type SRRR = (Y[3] - Y[2])/hRRR;
309  Z[0] = deriv2_4p_L( SR, hR, SRR, hRR, SRRR, hRRR );
310  } else {
311  real_type hR = X[1] - X[0];
312  real_type hRR = X[2] - X[1];
313  real_type hRRR = X[3] - X[2];
314  real_type hRRRR = X[4] - X[3];
315  real_type SR = (Y[1] - Y[0])/hR;
316  real_type SRR = (Y[2] - Y[1])/hRR;
317  real_type SRRR = (Y[3] - Y[2])/hRRR;
318  real_type SRRRR = (Y[4] - Y[3])/hRRRR;
319  Z[0] = deriv2_5p_L( SR, hR, SRR, hRR, SRRR, hRRR, SRRRR, hRRRR );
320  }
321  break;
322  case NATURAL_BC:
323  L[0] = 0; D[0] = 1; U[0] = 0; Z[0] = 0;
324  break;
325  case PARABOLIC_RUNOUT_BC:
326  L[0] = 0; D[0] = 1; U[0] = -1; Z[0] = 0;
327  break;
328  case NOT_A_KNOT:
329  {
330  real_type r = (X[1] - X[0])/(X[2] - X[1]);
331  // v0 - v1*(1+r) + r*v2 == 0
332  L[0] = 0;
333  D[0] = 1;
334  U[0] = -(1+r);
335  UU = r;
336  Z[0] = 0;
337  }
338  break;
339  }
340 
341  switch ( bcn ) {
342  case EXTRAPOLATE_BC:
343  L[n] = 0; D[n] = 1; U[n] = 0;
344  if ( npts == 2 ) {
345  Z[n] = 0;
346  } else if ( npts == 3 ) {
347  real_type hL = X[n] - X[n-1];
348  real_type hLL = X[n-1] - X[n-2];
349  real_type SL = (Y[n] - Y[n-1])/hL;
350  real_type SLL = (Y[n-1] - Y[n-2])/hLL;
351  Z[n] = deriv2_3p_R( SL, hL, SLL, hLL );
352  } else if ( npts == 4 ) {
353  real_type hL = X[n] - X[n-1];
354  real_type hLL = X[n-1] - X[n-2];
355  real_type hLLL = X[n-2] - X[n-3];
356  real_type SL = (Y[n] - Y[n-1])/hL;
357  real_type SLL = (Y[n-1] - Y[n-2])/hLL;
358  real_type SLLL = (Y[n-2] - Y[n-3])/hLLL;
359  Z[n] = deriv2_4p_R( SL, hL, SLL, hLL, SLLL, hLLL );
360  } else {
361  real_type hL = X[n] - X[n-1];
362  real_type hLL = X[n-1] - X[n-2];
363  real_type hLLL = X[n-2] - X[n-3];
364  real_type hLLLL = X[n-3] - X[n-4];
365  real_type SL = (Y[n] - Y[n-1])/hL;
366  real_type SLL = (Y[n-1] - Y[n-2])/hLL;
367  real_type SLLL = (Y[n-2] - Y[n-3])/hLLL;
368  real_type SLLLL = (Y[n-3] - Y[n-4])/hLLLL;
369  Z[n] = deriv2_5p_R( SL, hL, SLL, hLL, SLLL, hLLL, SLLLL, hLLLL );
370  }
371  break;
372  case NATURAL_BC:
373  L[n] = 0; D[n] = 1; U[n] = 0; Z[n] = 0;
374  break;
375  case PARABOLIC_RUNOUT_BC:
376  L[n] = -1; D[n] = 1; U[n] = 0; Z[n] = 0;
377  break;
378  case NOT_A_KNOT:
379  {
380  real_type r = (X[n-1] - X[n-2])/(X[n] - X[n-1]);
381  // r*v0 - v1*(1+r) + v2 == 0
382  U[n] = 0;
383  D[n] = 1;
384  L[n] = -(1+r);
385  LL = r;
386  Z[n] = 0;
387  }
388  break;
389  }
390 
391  if ( n > 2 ) {
392  Z[0] /= D[0];
393  U[0] /= D[0];
394  UU /= D[0];
395  D[1] -= L[1] * U[0];
396  U[1] -= L[1] * UU;
397  Z[1] -= L[1] * Z[0];
398  i = 1;
399  do {
400  Z[i] /= D[i];
401  U[i] /= D[i];
402  D[i+1] -= L[i+1] * U[i];
403  Z[i+1] -= L[i+1] * Z[i];
404  } while ( ++i < n );
405 
406  D[i] -= LL * U[i-2];
407  Z[i] -= LL * Z[i-2];
408 
409  Z[i] /= D[i];
410 
411  do {
412  --i;
413  Z[i] -= U[i] * Z[i+1];
414  } while ( i > 0 );
415 
416  Z[0] -= UU * Z[2];
417  }
418 
419  for ( i = 0; i < n; ++i ) {
420  real_type DX = X[i+1] - X[i];
421  Yp[i] = (Y[i+1]-Y[i])/DX - (2*Z[i] + Z[i+1]) * (DX/6);
422  }
423  real_type DX2 = (X[n] - X[n-1])/2;
424  Yp[n] = Yp[n-1] + DX2 * (Z[n-1] + Z[n]);
425 
426  }

434  real_type Yp[],

450  void
451  CubicSpline::build() {
452  UTILS_ASSERT(
453  m_npts > 1,
454  "CubicSpline::build(): npts = {} not enought points\n", m_npts
455  );
456  integer ibegin = 0;
457  integer iend = 0;
458  do {
459  // cerca intervallo monotono strettamente crescente
460  while ( ++iend < m_npts && m_X[iend-1] < m_X[iend] ) {}
461  CUBIC_SPLINE_TYPE_BC seg_bc0 = NOT_A_KNOT;
462  CUBIC_SPLINE_TYPE_BC seg_bcn = NOT_A_KNOT;
463  if ( ibegin == 0 ) seg_bc0 = m_bc0;
464  if ( iend == m_npts ) seg_bcn = m_bcn;
465  CubicSpline_build(
466  m_X+ibegin,
467  m_Y+ibegin,
468  m_Yp+ibegin,
469  iend - ibegin,
470  seg_bc0, seg_bcn
471  );
472  ibegin = iend;
473  } while ( iend < m_npts );
474 
475  SPLINE_CHECK_NAN( m_Yp, "CubicSpline::build(): Yp", m_npts );
476  }
