41  void
42  BiQuinticSpline::makeSpline() {
43 
44  size_t dim = size_t( m_nx*m_ny );
45  mem.allocate( 8*dim );
46  m_DX = mem( dim );
47  m_DY = mem( dim );
48  m_DXY = mem( dim );
49  m_DXX = mem( dim );
50  m_DYY = mem( dim );
51  m_DXYY = mem( dim );
52  m_DXXY = mem( dim );
53  m_DXXYY = mem( dim );
54 
55  // calcolo derivate
56  QuinticSpline sp, sp1;
57  //
58  for ( integer j = 0; j < m_ny; ++j ) {
59  sp.build( m_X, 1, &m_Z[size_t(this->ipos_C(0,j))], m_ny, m_nx );
60  for ( integer i = 0; i < m_nx; ++i ) {
61  size_t ij = size_t(this->ipos_C(i,j));
62  m_DX[ij] = sp.ypNode(i);
63  m_DXX[ij] = sp.yppNode(i);
64  }
65  }
66  for ( integer i = 0; i < m_nx; ++i ) {
67  sp.build( m_Y, 1, &m_Z[size_t(this->ipos_C(i,0))], 1, m_ny );
68  for ( integer j = 0; j < m_ny; ++j ) {
69  size_t ij = size_t(this->ipos_C(i,j));
70  m_DY[ij] = sp.ypNode(j);
71  m_DYY[ij] = sp.yppNode(j);
72  }
73  }
74  // interpolate derivative
75  for ( integer i = 0; i < m_nx; ++i ) {
76  sp.build( m_Y, 1, &m_DX[size_t(this->ipos_C(i,0))], 1, m_ny );
77  sp1.build( m_Y, 1, &m_DXX[size_t(this->ipos_C(i,0))], 1, m_ny );
78  for ( integer j = 0; j < m_ny; ++j ) {
79  size_t ij = size_t(this->ipos_C(i,j));
80  m_DXY[ij] = sp.ypNode(j);
81  m_DXYY[ij] = sp.yppNode(j);
82  m_DXXY[ij] = sp1.ypNode(j);
83  m_DXXYY[ij] = sp1.yppNode(j);
84  }
85  }
86  // interpolate derivative again
87  for ( integer j = 0; j < m_ny; ++j ) {
88  sp.build( m_X, 1, &m_DY[size_t(this->ipos_C(0,j))], m_ny, m_nx );
89  sp1.build( m_X, 1, &m_DYY[size_t(this->ipos_C(0,j))], m_ny, m_nx );
90  for ( integer i = 0; i < m_nx; ++i ) {
91  size_t ij = size_t(this->ipos_C(i,j));
92  m_DXY[ij] += sp.ypNode(i); m_DXY[ij] /= 2;
93  m_DXXY[ij] += sp.yppNode(i); m_DXXY[ij] /= 2;
94  m_DXYY[ij] += sp1.ypNode(i); m_DXYY[ij] /= 2;
95  m_DXXYY[ij] += sp1.yppNode(i); m_DXXYY[ij] /= 2;
96  }
97  }
98

100  //std::fill( DXX.begin(), DXX.end(), 0 );
101  //std::fill( DYY.begin(), DYY.end(), 0 );
102  }

113  size_t i01 = size_t(this->ipos_C(i-1,j,m_ny));
114  size_t i11 = size_t(this->ipos_C(i,j,m_ny));
115  fmt::print( s,
116  "patch({},{})\n"
117  " DX = {:<12.4} DY = {:<12.4}\n"
118  " Z00 = {:<12.4} Z01 = {:<12.4} Z10 = {:<12.4} Z11 = {:<12.4}\n"
119  " Dx00 = {:<12.4} Dx01 = {:<12.4} Dx10 = {:<12.4} Dx10 = {:<12.4}\n"
120  " Dy00 = {:<12.4} Dy01 = {:<12.4} Dy10 = {:<12.4} Dy11 = {:<12.4}\n",
121  i, j,
122  m_X[size_t(i)]-m_X[size_t(i-1)],
123  m_Y[size_t(j)]-m_Y[size_t(j-1)],
124  m_Z[i00], m_Z[i01], m_Z[i10], m_Z[i11],
125  m_DX[i00], m_DX[i01], m_DX[i10], m_DX[i11],
