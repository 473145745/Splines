41  void
42  BiCubicSpline::makeSpline() {
43  size_t nn = size_t(m_nx*m_ny);
44  m_mem_bicubic.allocate( 3*nn );
45  m_DX = m_mem_bicubic( nn );
46  m_DY = m_mem_bicubic( nn );
47  m_DXY = m_mem_bicubic( nn );
48 
49  // calcolo derivate
50  PchipSpline sp;
51  for ( integer j = 0; j < m_ny; ++j ) {
52  sp.build( m_X, 1, &m_Z[size_t(this->ipos_C(0,j))], m_ny, m_nx );
53  for ( integer i = 0; i < m_nx; ++i )
54  m_DX[size_t(this->ipos_C(i,j))] = sp.ypNode(i);
55  }
56  for ( integer i = 0; i < m_nx; ++i ) {
57  sp.build( m_Y, 1, &m_Z[size_t(this->ipos_C(i,0))], 1, m_ny );
58  for ( integer j = 0; j < m_ny; ++j )
59  m_DY[size_t(this->ipos_C(i,j))] = sp.ypNode(j);
60  }
61  std::fill_n( m_DXY, nn, 0 );
62  }

66  void
67  BiCubicSpline::writeToStream( ostream_type & s ) const {
68  fmt::print( "Nx = {} Ny = {}\n", m_nx, m_ny );
69  for ( integer i = 1; i < m_nx; ++i ) {
70  for ( integer j = 1; j < m_ny; ++j ) {
71  size_t i00 = size_t(this->ipos_C(i-1,j-1,m_ny));
72  size_t i10 = size_t(this->ipos_C(i,j-1,m_ny));
73  size_t i01 = size_t(this->ipos_C(i-1,j,m_ny));
74  size_t i11 = size_t(this->ipos_C(i,j,m_ny));
75  fmt::print( s,
76  "patch ({},{})\n"
77  "DX = {:<12.4} DY = {:<12.4}\n"
78  "Z00 = {:<12.4} Z01 = {:<12.4} Z10 = {:<12.4} Z11 = {:<12.4}\n"
79  "Dx00 = {:<12.4} Dx01 = {:<12.4} Dx10 = {:<12.4} Dx11 = {:<12.4}\n"
80  "Dy00 = {:<12.4} Dy01 = {:<12.4} Dy10 = {:<12.4} Dy11 = {:<12.4}\n",
81  i, j,
82  m_X[size_t(i)]-m_X[size_t(i-1)],
83  m_Y[size_t(j)]-m_Y[size_t(j-1)],
84  m_Z[i00], m_Z[i01], m_Z[i10], m_Z[i11],
85  m_DX[i00], m_DX[i01], m_DX[i10], m_DX[i11],
86  m_DY[i00], m_DY[i01], m_DY[i10], m_DY[i11]
87  );
88  }
89  }
90  }
