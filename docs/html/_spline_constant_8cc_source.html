56  void
57  ConstantSpline::reserve( integer n ) {
58  if ( m_external_alloc && n <= m_npts_reserved ) {
59  // nothing to do!, already allocated
60  } else {
61  m_baseValue.allocate( size_t(2*n) );
62  m_npts_reserved = n;
63  m_external_alloc = false;
64  m_X = m_baseValue( size_t(n) );
65  m_Y = m_baseValue( size_t(n) );
66  }
67  initLastInterval();
68  m_npts = 0;
69  }
