52  inline
53  real_type
54  Extrapolate3(
55  real_type X1,
56  real_type X2,
57  real_type X3,
58  real_type Z0,
59  real_type Z1,
60  real_type Z2
61  ) {
62  return ( (Z2-Z0) * (X3-X1)/X2 - (Z1-Z0) * (X3-X2)/X1 ) * (X3/(X2-X1)) + Z0;

67  static
68  void
69  estimate(
70  real_type z0,
71  real_type z1,
72  real_type z2,
73  real_type z3,
74  real_type x1,
75  real_type x2,
76  real_type x3,
77  real_type & c1,
78  real_type & c2,
79  real_type & c3,
80  real_type & SX,
81  real_type & SXX,
82  real_type & b0,
83  real_type & b1,
84  real_type RF[2],
85  real_type RI[2]
86  ) {
87 
88  // Primary estimate of partial derivative zx
89  // as the coefficient of the bicubic polynomial.
90  c1 = x2*x3/((x1-x2)*(x1-x3));
91  c2 = x3*x1/((x2-x3)*(x2-x1));
92  c3 = x1*x2/((x3-x1)*(x3-x2));
93  real_type primary_estimate = c1*(z1-z0)/x1 + c2*(z2-z0)/x2 + c3*(z3-z0)/x3;
94 
95  // Volatility factor and distance factor in the x direction
96  // for the primary estimate of zx.
97  SX = x1 + x2 + x3;
98  real_type SZ = z0 + z1 + z2 + z3;
99  SXX = x1*x1 + x2*x2 + x3*x3;
100  real_type SXZ = x1*z1 + z2*z2 + x3*z3;
101  real_type DNM = 4.0*SXX - SX*SX;
102  b0 = (SXX*SZ-SX*SXZ)/DNM;
103  b1 = (4.0*SXZ-SX*SZ)/DNM;
104  real_type dz0 = z0 - b0;
105  real_type dz1 = z1 - (b0+b1*x1);
106  real_type dz2 = z2 - (b0+b1*x2);
107  real_type dz3 = z3 - (b0+b1*x3);
108  real_type volatility_factor = dz0*dz0 + dz1*dz1 + dz2*dz2 + dz3*dz3;
109  // epsi value used to decide whether or not
110  // the volatility factor is essentially zero.
111  real_type epsi = (z0*z0+z1*z1+z2*z2+z3*z3)*1.0E-12;
112  // Accumulates the weighted primary estimates and their weights.
113  if ( volatility_factor > epsi ) { // finite weight.
114  real_type WT = 1.0/ (volatility_factor*SXX);
115  RF[1] += WT*primary_estimate;
116  RF[0] += WT;
117  } else { // infinite weight.
118  RI[1] += primary_estimate;
119  RI[0] += 1.0;
120  }
121  }

125  static
126  void
127  AkimaSmooth(
128  real_type const X[9], integer imin, integer imax,
129  real_type const Y[9], integer jmin, integer jmax,
130  real_type const Z[9][9],
131  real_type & DX,
132  real_type & DY,
133  real_type & DXY
134  ) {
135 
136  integer stencil[3][4] = {
137  { -3, -2, -1, 1 },
138  { -2, -1, 1, 2 },
139  { -1, 1, 2, 3 }
140  };
141 
142  real_type by0[4], by1[4], CY1A[4], CY2A[4], CY3A[4], SYA[4], SYYA[4];
143 
144  real_type X0 = X[4];
145  real_type Y0 = Y[4];
146  real_type Z00 = Z[4][4];
147 
148  // Initial setting
149  real_type DXF[2] = {0,0};
150  real_type DXI[2] = {0,0};
151  real_type DYF[2] = {0,0};
152  real_type DYI[2] = {0,0};
153  real_type DXYF[2] = {0,0};
154  real_type DXYI[2] = {0,0};
155 
156  for ( integer k = 0; k < 4; ++k ) {
157  int j1 = 4 + stencil[0][k];
158  int j2 = 4 + stencil[1][k];
159  int j3 = 4 + stencil[2][k];
160  if ( j1 >= int(jmin) && j3 <= int(jmax) )
161  estimate(
162  Z00, Z[4][j1], Z[4][j2], Z[4][j3],
163  Y[j1] - Y0, Y[j2] - Y0, Y[j3] - Y0,
164  CY1A[k], CY2A[k], CY3A[k], SYA[k], SYYA[k],
165  by0[k], by1[k], DYF, DYI
166  );
167  }
168 
169  for ( integer kx = 0; kx < 4; ++kx ) {
170  int i1 = 4 + stencil[0][kx];
171  int i2 = 4 + stencil[1][kx];
172  int i3 = 4 + stencil[2][kx];
173 
174  if ( i1 < int(imin) || i3 > int(imax) ) continue;
175 
176  real_type X1 = X[i1] - X0;
177  real_type X2 = X[i2] - X0;
178  real_type X3 = X[i3] - X0;
179  real_type Z10 = Z[i1][4];
180  real_type Z20 = Z[i2][4];
181  real_type Z30 = Z[i3][4];
182  real_type CX1, CX2, CX3, SX, SXX, B00X, B10;
183  estimate(
184  Z00, Z10, Z20, Z30,
185  X1, X2, X3,
186  CX1, CX2, CX3,
187  SX, SXX,
188  B00X, B10, DXF, DXI
189  );
190 
191  for ( int ky = 0; ky < 4; ++ky ) {
192  int j1 = 4 + stencil[0][ky];
193  int j2 = 4 + stencil[1][ky];
194  int j3 = 4 + stencil[2][ky];
195  if ( j1 < int(jmin) || j3 > int(jmax) ) continue;
196 
197  real_type Y1 = Y[j1] - Y0;
198  real_type Y2 = Y[j2] - Y0;
199  real_type Y3 = Y[j3] - Y0;
200  real_type CY1 = CY1A[ky];
201  real_type CY2 = CY2A[ky];
202  real_type CY3 = CY3A[ky];
203  real_type SY = SYA[ky];
204  real_type SYY = SYYA[ky];
205  real_type B00Y = by0[ky];
206  real_type B01 = by1[ky];
207 
208  real_type Z01 = Z[4][j1], Z02 = Z[4][j2], Z03 = Z[4][j3];
209  real_type Z11 = Z[i1][j1], Z12 = Z[i1][j2], Z13 = Z[i1][j3];
210  real_type Z21 = Z[i2][j1], Z22 = Z[i2][j2], Z23 = Z[i2][j3];
211  real_type Z31 = Z[i3][j1], Z32 = Z[i3][j2], Z33 = Z[i3][j3];
212 
213  // Primary estimate of partial derivative zxy
214  // as the coefficient of the bicubic polynomial.
215  real_type DZXY11 = (Z11-Z10-Z01+Z00)/(X1*Y1);
216  real_type DZXY12 = (Z12-Z10-Z02+Z00)/(X1*Y2);
217  real_type DZXY13 = (Z13-Z10-Z03+Z00)/(X1*Y3);
218  real_type DZXY21 = (Z21-Z20-Z01+Z00)/(X2*Y1);
219  real_type DZXY22 = (Z22-Z20-Z02+Z00)/(X2*Y2);
220  real_type DZXY23 = (Z23-Z20-Z03+Z00)/(X2*Y3);
221  real_type DZXY31 = (Z31-Z30-Z01+Z00)/(X3*Y1);
222  real_type DZXY32 = (Z32-Z30-Z02+Z00)/(X3*Y2);
223  real_type DZXY33 = (Z33-Z30-Z03+Z00)/(X3*Y3);
224  real_type PEZXY = CX1* (CY1*DZXY11+CY2*DZXY12+CY3*DZXY13) +
225  CX2* (CY1*DZXY21+CY2*DZXY22+CY3*DZXY23) +
226  CX3* (CY1*DZXY31+CY2*DZXY32+CY3*DZXY33);
227  // Volatility factor and distance factor
228  // in the x and y directions for the primary estimate of zxy.
229  real_type B00 = (B00X+B00Y)/2.0;
230  real_type SXY = SX*SY;
231  real_type SXXY = SXX*SY;
232  real_type SXYY = SX*SYY;
233  real_type SXXYY = SXX*SYY;
234  real_type SXYZ = X1 * (Y1*Z11+Y2*Z12+Y3*Z13) +
235  X2 * (Y1*Z21+Y2*Z22+Y3*Z23) +
236  X3 * (Y1*Z31+Y2*Z32+Y3*Z33);
237  real_type B11 = (SXYZ-B00*SXY-B10*SXXY-B01*SXYY)/SXXYY;
238  real_type DZ00 = Z00 - B00;
239  real_type DZ01 = Z01 - (B00+B01*Y1);
240  real_type DZ02 = Z02 - (B00+B01*Y2);
241  real_type DZ03 = Z03 - (B00+B01*Y3);
242  real_type DZ10 = Z10 - (B00+B10*X1);
243  real_type DZ11 = Z11 - (B00+B01*Y1+X1*(B10+B11*Y1));
244  real_type DZ12 = Z12 - (B00+B01*Y2+X1*(B10+B11*Y2));
245  real_type DZ13 = Z13 - (B00+B01*Y3+X1*(B10+B11*Y3));
246  real_type DZ20 = Z20 - (B00+B10*X2);
247  real_type DZ21 = Z21 - (B00+B01*Y1+X2*(B10+B11*Y1));
248  real_type DZ22 = Z22 - (B00+B01*Y2+X2*(B10+B11*Y2));
249  real_type DZ23 = Z23 - (B00+B01*Y3+X2*(B10+B11*Y3));
250  real_type DZ30 = Z30 - (B00+B10*X3);
251  real_type DZ31 = Z31 - (B00+B01*Y1+X3*(B10+B11*Y1));
252  real_type DZ32 = Z32 - (B00+B01*Y2+X3*(B10+B11*Y2));
253  real_type DZ33 = Z33 - (B00+B01*Y3+X3*(B10+B11*Y3));
254  real_type volatility_factor = DZ00*DZ00 + DZ01*DZ01 +
255  DZ02*DZ02 + DZ03*DZ03 +
256  DZ10*DZ10 + DZ11*DZ11 +
257  DZ12*DZ12 + DZ13*DZ13 +
258  DZ20*DZ20 + DZ21*DZ21 +
259  DZ22*DZ22 + DZ23*DZ23 +
260  DZ30*DZ30 + DZ31*DZ31 +
261  DZ32*DZ32 + DZ33*DZ33;
262  real_type epsi = (Z00*Z00+Z01*Z01+Z02*Z02+Z03*Z03+
263  Z10*Z10+Z11*Z11+Z12*Z12+Z13*Z13+
264  Z20*Z20+Z21*Z21+Z22*Z22+Z23*Z23+
265  Z30*Z30+Z31*Z31+Z32*Z32+Z33*Z33)*1.0E-12;
266  // Accumulates the weighted primary estimates of zxy and their weights.
267  if ( volatility_factor > epsi ) { // finite weight.
268  real_type WT = 1 / (volatility_factor*SXX*SYY);
269  DXYF[1] += WT*PEZXY;
270  DXYF[0] += WT;
271  } else { // infinite weight.
272  DXYI[1] += PEZXY;
273  DXYI[0] += 1.0;
274  }
275  }
276  }
277  DX = DXI[0] < 0.5 ? DXF[1]/DXF[0] : DXI[1]/DXI[0];
278  DY = DYI[0] < 0.5 ? DYF[1]/DYF[0] : DYI[1]/DYI[0];
279  DXY = DXYI[0] < 0.5 ? DXYF[1]/DXYF[0] : DXYI[1]/DXYI[0];
280  }
281 
282  /*
283  * This subroutine estimates three partial derivatives, zx, zy, and
284  * zxy, of a bivariate function, z(x,y), on a rectangular grid in
285  * the x-y plane. It is based on the revised Akima method that has
286  * the accuracy of a bicubic polynomial.
287  */
288  void
289  Akima2Dspline::makeSpline() {
290 
291  size_t nn = size_t( m_nx*m_ny );
292  m_mem_bicubic.allocate( 3*nn );
293  m_DX = m_mem_bicubic( nn );
294  m_DY = m_mem_bicubic( nn );
295  m_DXY = m_mem_bicubic( nn );
296 
297  std::fill_n(m_DX,nn,0);
298  std::fill_n(m_DY,nn,0);
299  std::fill_n(m_DXY,nn,0);
300 
301  real_type x_loc[9], y_loc[9], z_loc[9][9];
302 
303  for ( size_t i0 = 0; i0 < size_t(m_nx); ++i0 ) {
304  size_t imin = 4 > i0 ? 4-i0 : 0;
305  size_t imax = size_t(m_nx) < 5+i0 ? 3+(size_t(m_nx)-i0) : 8;
306 
307  for ( size_t i = imin; i <= imax; ++i ) x_loc[i] = m_X[i+i0-4]-m_X[i0];
308 
309  for ( size_t j0 = 0; j0 < size_t(m_ny); ++j0 ) {
310  size_t jmin = 4 > j0 ? 4-j0 : 0;
311  size_t jmax = size_t(m_ny) < 5+j0 ? 3+(size_t(m_ny)-j0) : 8;
312 
313  for ( size_t j = jmin; j <= jmax; ++j ) y_loc[j] = m_Y[j+j0-4]-m_Y[j0];
314 
315  for ( size_t i = imin; i <= imax; ++i )
316  for ( size_t j = jmin; j <= jmax; ++j )
317  z_loc[i][j] = m_Z[size_t(ipos_C(integer(i+i0-4),
318  integer(j+j0-4),
319  m_ny))];
320 
321  // if not enough points, extrapolate
322  size_t iadd = 0, jadd = 0;
323  if ( imax < 3+imin ) {
324  x_loc[imin-1] = 2*x_loc[imin] - x_loc[imax];
325  x_loc[imax+1] = 2*x_loc[imax] - x_loc[imin];
326  iadd = 1;
327  if ( imax == 1+imin ) {
328  real_type x0 = x_loc[imin];
329  real_type x1 = x_loc[imax];
330  for ( size_t j = jmin; j <= jmax; ++j ) {
331  real_type z0 = z_loc[imin][j];
332  real_type z1 = z_loc[imax][j];
333  z_loc[imin-1][j] = Extrapolate2( x0-x1, x_loc[imin-1]-x1, z1, z0 );
334  z_loc[imax+1][j] = Extrapolate2( x1-x0, x_loc[imax+1]-x0, z0, z1 );
335  }
336  } else {
337  real_type x0 = x_loc[imin];
338  real_type x1 = x_loc[imin+1];
339  real_type x2 = x_loc[imax];
340  for ( size_t j = jmin; j <= jmax; ++j ) {
341  real_type z0 = z_loc[imin][j];
342  real_type z1 = z_loc[imin+1][j];
343  real_type z2 = z_loc[imax][j];
344  z_loc[imin-1][j] = Extrapolate3( x1-x2, x0-x2, x_loc[imin-1]-x1, z2, z1, z0 );
345  z_loc[imax+1][j] = Extrapolate3( x1-x0, x2-x0, x_loc[imax+1]-x0, z0, z1, z2 );
346  }
347  }
348  }
349  if ( jmax < 3+jmin ) {
350  y_loc[jmin-1] = 2*y_loc[jmin] - y_loc[jmax];
351  y_loc[jmax+1] = 2*y_loc[jmax] - y_loc[jmin];
352  jadd = 1;
353  if ( jmax-jmin == 1 ) {
354  real_type y0 = y_loc[jmin];
355  real_type y1 = y_loc[jmax];
356  for ( size_t i = imin-iadd; i <= imax+iadd; ++i ) {
357  real_type z0 = z_loc[i][jmin];
358  real_type z1 = z_loc[i][jmax];
359  z_loc[i][jmin-1] = Extrapolate2( y0-y1, y_loc[jmin-1]-y1, z1, z0 );
360  z_loc[i][jmax+1] = Extrapolate2( y1-y0, y_loc[jmax+1]-y0, z0, z1 );
361  }
362  } else {
363  real_type y0 = y_loc[jmin];
364  real_type y1 = y_loc[jmin+1];
365  real_type y2 = y_loc[jmax];
366  for ( size_t i = imin-iadd; i <= imax+iadd; ++i ) {
367  real_type z0 = z_loc[i][jmin];
368  real_type z1 = z_loc[i][jmin+1];
369  real_type z2 = z_loc[i][jmax];
370  z_loc[i][imin-1] = Extrapolate3( y1-y2, y0-y2, y_loc[jmin-1]-y1, z2, z1, z0 );
371  z_loc[i][imax+1] = Extrapolate3( y1-y0, y2-y0, y_loc[jmax+1]-y0, z0, z1, z2 );
372  }
373  }
374  }
375 
376  size_t i0j0 = size_t(ipos_C(integer(i0),integer(j0)));
377 
378  AkimaSmooth( x_loc, integer(imin-iadd), integer(imax+iadd),
379  y_loc, integer(jmin-jadd), integer(jmax+jadd),
380  z_loc, m_DX[i0j0], m_DY[i0j0], m_DXY[i0j0] );
381  }
382  }
383  }
