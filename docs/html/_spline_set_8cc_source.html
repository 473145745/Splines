45  integer
46  SplineSet::BinarySearch::search( std::string const & id ) const {\
47  //for ( auto e : data )
48  // std::cout << e.first << " -> " << e.second << '\n';
49  // binary search
50  size_t U = data.size();
51  size_t L = 0;
52  while ( U-L > 1 ) {
53  size_t pos = (L+U)>>1;
54  std::string const & id_pos = data[pos].first;
55  if ( id_pos < id ) L = pos; else U = pos;
56  }
57  if ( data[L].first == id ) return data[L].second;
58  if ( data[U].first == id ) return data[U].second;
59  return -1; // non trovato
60  }

64  void
65  SplineSet::BinarySearch::insert( std::string const & id, integer position ) {
66  size_t pos = data.size();
67  data.push_back(DATA_TYPE(id,position));
68  while ( pos > 0 ) {
69  size_t pos1 = pos-1;
70  data[pos].first = data[pos1].first;
71  data[pos].second = data[pos1].second;
72  if ( data[pos1].first < id ) break;
73  pos = pos1;
74  }
75  data[pos] = DATA_TYPE(id,position);
76  }

212  void
213  SplineSet::build(
214  integer nspl,
215  integer npts,
216  char const *headers[],
217  SplineType1D const stype[],
218  real_type const X[],
219  real_type const *Y[],
220  real_type const *Yp[]
221  ) {
222  UTILS_ASSERT(
223  nspl > 0,
224  "SplineSet::build expected positive nspl = {}\n", nspl
225  );
226  UTILS_ASSERT(
227  npts > 1,
228  "SplineSet::build expected npts = {} greather than 1", npts
229  );
230  m_nspl = nspl;
231  m_npts = npts;
232  // allocate memory
233  m_baseSplines.allocate( size_t(m_nspl) );
234  m_splines = reinterpret_cast<Spline **>(m_baseSplines( size_t(m_nspl) ));
235 
236  m_baseInt.allocate( size_t(m_nspl) );
237  m_is_monotone = m_baseInt( size_t(m_nspl) );
238 
239  m_header_to_position.clear();
240 
241  integer mem = npts;
242  for ( integer spl = 0; spl < nspl; ++spl ) {
243  switch (stype[size_t(spl)]) {
244  case QUINTIC_TYPE:
245  mem += npts; // Y, Yp, Ypp
246  case CUBIC_TYPE:
247  case AKIMA_TYPE:
248  case BESSEL_TYPE:
249  case PCHIP_TYPE:
250  case HERMITE_TYPE:
251  mem += npts; // Y, Yp
252  case CONSTANT_TYPE:
253  case LINEAR_TYPE:
254  mem += npts;
255  break;
256  case SPLINE_SET_TYPE:
257  case SPLINE_VEC_TYPE:
258  //default:
259  UTILS_ERROR(
260  "SplineSet::build\nAt spline n.{} named {} cannot be done for type = {}\n",
261  spl, headers[spl], stype[spl]
262  );
263  }
264  }
265 
266  m_baseValue.allocate( size_t(mem + 2*nspl) );
267  m_basePointer.allocate( size_t(3*nspl) );
268 
269  m_Y = m_basePointer ( size_t(m_nspl) );
270  m_Yp = m_basePointer ( size_t(m_nspl) );
271  m_Ypp = m_basePointer ( size_t(m_nspl) );
272  m_X = m_baseValue ( size_t(m_npts) );
273  m_Ymin = m_baseValue ( size_t(m_nspl) );
274  m_Ymax = m_baseValue ( size_t(m_nspl) );
275 
276  std::copy_n( X, npts, m_X );
277  for ( size_t spl = 0; spl < size_t(nspl); ++spl ) {
278  real_type *& pY = m_Y[spl];
279  real_type *& pYp = m_Yp[spl];
280  real_type *& pYpp = m_Ypp[spl];
281  pY = m_baseValue(size_t(m_npts));
282  std::copy_n( Y[spl], npts, pY );
283  if ( stype[spl] == CONSTANT_TYPE ) {
284  m_Ymin[spl] = *std::min_element( pY, pY+npts-1 );
285  m_Ymax[spl] = *std::max_element( pY, pY+npts-1 );
286  } else {
287  m_Ymin[spl] = *std::min_element( pY, pY+npts );
288  m_Ymax[spl] = *std::max_element( pY, pY+npts );
289  }
290  pYpp = pYp = nullptr;
291  switch ( stype[size_t(spl)] ) {
292  case QUINTIC_TYPE:
293  pYpp = m_baseValue( size_t(m_npts) );
294  case CUBIC_TYPE:
295  case AKIMA_TYPE:
296  case BESSEL_TYPE:
297  case PCHIP_TYPE:
298  case HERMITE_TYPE:
299  pYp = m_baseValue( size_t(m_npts) );
300  if ( stype[spl] == HERMITE_TYPE ) {
301  UTILS_ASSERT(
302  Yp != nullptr && Yp[spl] != nullptr,
303  "SplineSet::build\nAt spline n.{} named {}\n"
304  "expect to find derivative values",
305  spl, headers[spl]
306  );
307  std::copy_n( Yp[spl], npts, pYp );
308  }
309  case CONSTANT_TYPE:
310  case LINEAR_TYPE:
311  case SPLINE_SET_TYPE:
312  case SPLINE_VEC_TYPE:
313  //default:
314  break;
315  }
316  string h = headers[spl];
317  Spline * & s = m_splines[spl];
318 
319  m_is_monotone[spl] = -1;
320  switch (stype[size_t(spl)]) {
321  case CONSTANT_TYPE:
322  s = new ConstantSpline(h);
323  static_cast<ConstantSpline*>(s)->reserve_external( m_npts, m_X, pY );
324  static_cast<ConstantSpline*>(s)->m_npts = m_npts;
325  static_cast<ConstantSpline*>(s)->build();
326  break;
327 
328  case LINEAR_TYPE:
329  s = new LinearSpline(h);
330  static_cast<LinearSpline*>(s)->reserve_external( m_npts, m_X, pY );
331  static_cast<LinearSpline*>(s)->m_npts = m_npts;
332  static_cast<LinearSpline*>(s)->build();
333  // check monotonicity of data
334  { integer flag = 1;
335  for ( integer j = 1; j < m_npts; ++j ) {
336  if ( pY[j-1] > pY[j] ) { flag = -1; break; } // non monotone data
337  if ( Utils::isZero(pY[j-1]-pY[j]) && m_X[j-1] < m_X[j] ) flag = 0; // non strict monotone
338  }
339  m_is_monotone[spl] = flag;
340  }
341  break;
342 
343  case CUBIC_TYPE:
344  s = new CubicSpline(h);
345  static_cast<CubicSpline*>(s)->reserve_external( m_npts, m_X, pY, pYp );
346  static_cast<CubicSpline*>(s)->m_npts = m_npts;
347  static_cast<CubicSpline*>(s)->build();
348  m_is_monotone[spl] = checkCubicSplineMonotonicity( m_X, pY, pYp, m_npts );
349  break;
350 
351  case AKIMA_TYPE:
352  s = new AkimaSpline(h);
353  static_cast<AkimaSpline*>(s)->reserve_external( m_npts, m_X, pY, pYp );
354  static_cast<AkimaSpline*>(s)->m_npts = m_npts;
355  static_cast<AkimaSpline*>(s)->build();
356  m_is_monotone[spl] = checkCubicSplineMonotonicity( m_X, pY, pYp, m_npts );
357  break;
358 
359  case BESSEL_TYPE:
360  s = new BesselSpline(h);
361  static_cast<BesselSpline*>(s)->reserve_external( m_npts, m_X, pY, pYp );
362  static_cast<BesselSpline*>(s)->m_npts = m_npts;
363  static_cast<BesselSpline*>(s)->build();
364  m_is_monotone[spl] = checkCubicSplineMonotonicity( m_X, pY, pYp, m_npts );
365  break;
366 
367  case PCHIP_TYPE:
368  s = new PchipSpline(h);
369  static_cast<PchipSpline*>(s)->reserve_external( m_npts, m_X, pY, pYp );
370  static_cast<PchipSpline*>(s)->m_npts = m_npts;
371  static_cast<PchipSpline*>(s)->build();
372  m_is_monotone[spl] = checkCubicSplineMonotonicity( m_X, pY, pYp, m_npts );
373  break;
374 
375  case HERMITE_TYPE:
376  s = new HermiteSpline(h);
377  static_cast<CubicSpline*>(s)->reserve_external( m_npts, m_X, pY, pYp );
378  static_cast<CubicSpline*>(s)->m_npts = m_npts;
379  static_cast<CubicSpline*>(s)->build();
380  m_is_monotone[spl] = checkCubicSplineMonotonicity( m_X, pY, pYp, m_npts );
381  break;
382 
383  case QUINTIC_TYPE:
384  s = new QuinticSpline(h);
385  static_cast<QuinticSpline*>(s)->reserve_external( m_npts, m_X, pY, pYp, pYpp );
386  static_cast<QuinticSpline*>(s)->m_npts = m_npts;
387  static_cast<QuinticSpline*>(s)->build();
388  break;
389 
390  case SPLINE_SET_TYPE:
391  case SPLINE_VEC_TYPE:
392  //default:
393  UTILS_ERROR(
394  "SplineSet::build\nAt spline n.{} named {}\n"
395  "{} not allowed as spline type\n"
396  "in SplineSet::build for {}-th spline\n",
397  spl, headers[spl], stype[size_t(spl)], spl
398  );
399  }
400  m_header_to_position.insert( s->name(), integer(spl) );
401  }
402 
403  m_baseValue . must_be_empty( "SplineSet::build, baseValue" );
404  m_basePointer . must_be_empty( "SplineSet::build, basePointer" );
405 
406  }

491 
492  Spline const *
493  SplineSet::intersect(
494  integer spl,
495  real_type zeta,
496  real_type & x
497  ) const {
498  UTILS_ASSERT(
499  spl >= 0 && spl < m_nspl,
500  "Spline n.{} is not in SplineSet", spl
501  );
502  UTILS_ASSERT(
503  m_is_monotone[size_t(spl)]>0,
504  "Spline n.{} is not monotone and can't be used as independent", spl
505  );
506  Spline const * S = m_splines[spl];
507  // cerco intervallo intersezione
508  real_type const * X = m_Y[spl];
509  UTILS_ASSERT(
510  zeta >= X[0] && zeta <= X[m_npts-1],
511  "SplineSet, evaluation at zeta = {} is out of range: [{},{}]\n",
512  zeta, X[0], X[m_npts-1]
513  );
514 
515  integer interval = integer(lower_bound( X, X+m_npts, zeta ) - X);
516  if ( interval > 0 ) --interval;
517  if ( Utils::isZero(X[size_t(interval)]-X[size_t(interval+1)]) ) ++interval; // degenerate interval for duplicated nodes
518  if ( interval >= m_npts-1 ) interval = m_npts-2;
519 
520  // compute intersection
521  real_type a = m_X[interval];
522  real_type b = m_X[interval+1];
523  real_type ya = X[interval];
524  real_type yb = X[interval+1];
525  real_type DX = b-a;
526  real_type DY = yb-ya;
527  UTILS_ASSERT(
528  zeta >= ya && zeta <= yb,
529  "SplineSet, Bad interval [{},{}] for zeta = {}\n", ya, yb, zeta
530  );
531  UTILS_ASSERT(
532  a < b,
533  "SplineSet, Bad x interval [{},{}]\n", a, b
534  );
535  if ( S->type() == LINEAR_TYPE ) {
536  x = a + (b-a)*(zeta-ya)/(yb-ya);
537  } else {
538  real_type const * dX = m_Yp[spl];
539  real_type dya = dX[interval];
540  real_type dyb = dX[interval+1];
541  PolynomialRoots::Cubic cubic(
542  (dyb+dya-2*DY/DX)/(DX*DX), // A
543  (3*DY/DX-2*dya-dyb)/DX, // B
544  dya, // C
545  ya-zeta // D
546  );
547  real_type r[3];
548  integer npr = cubic.getRealRoots( r );
549  // cerca radice buona
550  bool ok = false;
551  for ( integer i = 0; i < npr && !ok; ++i ) {
552  ok = r[i] >= 0 && r[i] <= DX;
553  if ( ok ) x = a + r[i];
554  }
555  UTILS_ASSERT(
556  ok,
557  "SplineSet, failed to find intersection with independent spline at zeta = {}\n", zeta
558  );
559  }
560  return S;
561  }
