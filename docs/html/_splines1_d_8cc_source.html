89  tp = CONSTANT_TYPE;
90  } else if ( spl_type == "linear" ) {
91  tp = LINEAR_TYPE;
92  } else if ( spl_type == "cubic" ) {
93  tp = CUBIC_TYPE;
94  } else if ( spl_type == "akima" ) {
95  tp = AKIMA_TYPE;
96  } else if ( spl_type == "bessel" ) {
97  tp = BESSEL_TYPE;
98  } else if ( spl_type == "pchip" ) {
99  tp = PCHIP_TYPE;
100  } else if ( spl_type == "quintic" ) {
101  tp = QUINTIC_TYPE;
102  } else {
103  UTILS_ERROR(
104  "Spline1D::setup[{}] unknown type {}, not in [constant,linear,cubic,akima,bessel,pchip,quintic]\n",
105  m_name, spl_type
106  );
107  }
108  if ( m_pSpline != nullptr ) delete m_pSpline;
109  m_pSpline = new_Spline1D( m_name, tp );
110  m_pSpline->build( gc );
111  }
