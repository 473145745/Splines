63  void
64  SplineSurf::build(
65  real_type const x[], integer incx,
66  real_type const y[], integer incy,
67  real_type const z[], integer ldZ,
68  integer nx, integer ny,
69  bool fortran_storage,
70  bool transposed
71  ) {
72  m_nx = nx;
73  m_ny = ny;
74  m_mem.allocate( size_t((nx+1)*(ny+1)) );
75  m_X = m_mem( size_t(nx) );
76  m_Y = m_mem( size_t(ny) );
77  m_Z = m_mem( size_t(nx*ny) );
78  for ( size_t i = 0; i < size_t(nx); ++i ) m_X[i] = x[i*size_t(incx)];
79  for ( size_t i = 0; i < size_t(ny); ++i ) m_Y[i] = y[i*size_t(incy)];
80  if ( (fortran_storage && transposed) || (!fortran_storage && !transposed) ) {
81  UTILS_ASSERT(
82  ldZ >= ny,
83  "SplineSurf::build, ldZ = {} must be >= of nx = {}\n",
84  ldZ, ny
85  );
86  for ( integer i = 0; i < nx; ++i )
87  for ( integer j = 0; j < ny; ++j )
88  m_Z[size_t(ipos_C(i,j,ny))] = z[size_t(ipos_C(i,j,ldZ))];
89  } else {
90  UTILS_ASSERT(
91  ldZ >= nx,
92  "SplineSurf::build, ldZ = {} must be >= of ny = {}\n",
93  ldZ, nx
94  );
95  for ( integer i = 0; i < nx; ++i )
96  for ( integer j = 0; j < ny; ++j )
97  m_Z[size_t(ipos_C(i,j,ny))] = z[size_t(ipos_F(i,j,ldZ))];
98  }
99  m_Z_max = *std::max_element(m_Z,m_Z+nx*ny);
100  m_Z_min = *std::min_element(m_Z,m_Z+nx*ny);
101  makeSpline();
102  }

107  SplineSurf::build(
108  real_type const z[], integer ldZ,
109  integer nx, integer ny,
110  bool fortran_storage,
111  bool transposed
112  ) {

114  mem.allocate( size_t(nx+ny) );
115  real_type * XX = mem( size_t(nx) );
116  real_type * YY = mem( size_t(ny) ); // temporary vector
117  for ( size_t i = 0; i < size_t(nx); ++i ) XX[i] = real_type(i);
118  for ( size_t i = 0; i < size_t(ny); ++i ) YY[i] = real_type(i);
119  build( XX, 1, YY, 1, z, ldZ, nx, ny, fortran_storage, transposed );
120  }

125  BiCubicSplineBase::load(
126  integer i, integer j, real_type bili3[4][4]
127  ) const {
128  //
129  // 1 3
130  //
131  // 0 2
132  //
133  size_t i0 = size_t(ipos_C(i,j,m_ny));
134  size_t i1 = size_t(ipos_C(i,j+1,m_ny));
135  size_t i2 = size_t(ipos_C(i+1,j,m_ny));
136  size_t i3 = size_t(ipos_C(i+1,j+1,m_ny));
137 
138  bili3[0][0] = m_Z[i0]; bili3[0][1] = m_Z[i1];
139  bili3[0][2] = m_DY[i0]; bili3[0][3] = m_DY[i1];
140 
141  bili3[1][0] = m_Z[i2]; bili3[1][1] = m_Z[i3];
142  bili3[1][2] = m_DY[i2]; bili3[1][3] = m_DY[i3];
143 
144  bili3[2][0] = m_DX[i0]; bili3[2][1] = m_DX[i1];
145  bili3[2][2] = m_DXY[i0]; bili3[2][3] = m_DXY[i1];
146 
147  bili3[3][0] = m_DX[i2]; bili3[3][1] = m_DX[i3];
148  bili3[3][2] = m_DXY[i2]; bili3[3][3] = m_DXY[i3];
149  }

160  load( i, j, bili3 );
161  return bilinear3( u, bili3, v );

248  void
249  BiCubicSplineBase::DD( real_type x, real_type y, real_type d[6] ) const {
250  real_type bili3[4][4], u[4], u_D[4], u_DD[4], v[3], v_D[4], v_DD[4];
251  integer i = search_x( x );
252  integer j = search_y( y );
253  Hermite3 ( x - m_X[size_t(i)], m_X[size_t(i+1)] - m_X[size_t(i)], u );
254  Hermite3_D ( x - m_X[size_t(i)], m_X[size_t(i+1)] - m_X[size_t(i)], u_D );
255  Hermite3_DD( x - m_X[size_t(i)], m_X[size_t(i+1)] - m_X[size_t(i)], u_DD );
256  Hermite3 ( y - m_Y[size_t(j)], m_Y[size_t(j+1)] - m_Y[size_t(j)], v );
257  Hermite3_D ( y - m_Y[size_t(j)], m_Y[size_t(j+1)] - m_Y[size_t(j)], v_D );
258  Hermite3_DD( y - m_Y[size_t(j)], m_Y[size_t(j+1)] - m_Y[size_t(j)], v_DD );
259  load( i, j, bili3 );
260  d[0] = bilinear3( u, bili3, v );
261  d[1] = bilinear3( u_D, bili3, v );
262  d[2] = bilinear3( u, bili3, v_D );
263  d[3] = bilinear3( u_DD, bili3, v );
264  d[4] = bilinear3( u_D, bili3, v_D );
265  d[5] = bilinear3( u, bili3, v_DD );
266  }

276  size_t i01 = size_t(ipos_C(i,j+1,m_ny));

279 
280  //
281  // 1 3
282  //
283  // 0 2
284  //
285  // H0, H1, dH0, dH1, ddH0, ddH1
286 
287  // + + + + + +
288  // + + + + + +
289  // + + + + . .
290  // + + + + . .
291  // + + . . . .
292  // + + . . . .
293 
294  // P
295  bili5[0][0] = m_Z[i00]; bili5[0][1] = m_Z[i01];
296  bili5[1][0] = m_Z[i10]; bili5[1][1] = m_Z[i11];
297 
298  // DX
299  bili5[2][0] = m_DX[i00]; bili5[2][1] = m_DX[i01];
300  bili5[3][0] = m_DX[i10]; bili5[3][1] = m_DX[i11];
301 
302  // DXX
303  bili5[4][0] = m_DXX[i00]; bili5[4][1] = m_DXX[i01];
304  bili5[5][0] = m_DXX[i10]; bili5[5][1] = m_DXX[i11];
305 
306  // DY
307  bili5[0][2] = m_DY[i00]; bili5[0][3] = m_DY[i01];
308  bili5[1][2] = m_DY[i10]; bili5[1][3] = m_DY[i11];
309 
310  // DYY
311  bili5[0][4] = m_DYY[i00]; bili5[0][5] = m_DYY[i01];
312  bili5[1][4] = m_DYY[i10]; bili5[1][5] = m_DYY[i11];
313 
314  // DXY
315  bili5[2][2] = m_DXY[i00]; bili5[2][3] = m_DXY[i01];
316  bili5[3][2] = m_DXY[i10]; bili5[3][3] = m_DXY[i11];
317 
318  // DXXY
319  bili5[4][2] = m_DXXY[i00]; bili5[4][3] = m_DXXY[i01];
320  bili5[5][2] = m_DXXY[i10]; bili5[5][3] = m_DXXY[i11];
321 
322  // DXYY
323  bili5[2][4] = m_DXYY[i00]; bili5[2][5] = m_DXYY[i01];
324  bili5[3][4] = m_DXYY[i10]; bili5[3][5] = m_DXYY[i11];

327  bili5[4][4] = m_DXXYY[i00]; bili5[4][5] = m_DXXYY[i01];
